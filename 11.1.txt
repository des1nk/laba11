//Даны целые положительные числа A и B (A < B). Вывести все целые числа от A до B включительно; при этом каждое число должно выводиться столько раз, каково его значение (например, число 3 выводится 3 раза).
#include <iostream>
#include <string>
using namespace std;
int main()
{
  int A, B, x;
  cout << "A=";
  cin >> A;
  cout << "B=";
  cin >> B;
  if (A < B)
  {
   for (A; A<=B; A=A+1)
   {
       x=A;
       while (x>0)
       {
           cout << A;
           x=x-1;
       }
   }
  }
}
