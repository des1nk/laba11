//Дано целое число N (> 1). Вывести наименьшее из целых чисел K, для которых сумма 1 + 2 + . . . + K будет больше или равна N, и саму эту сумму.
#include <iostream>
#include <string>
using namespace std;
int main()
{
  int N, i, K;
  cout << "N=";
  cin >> N;
  if ( N>1 )
  {
  K=0;
  for (i=1;K<N ; i=i+1)
  {
      K=K+i;
  }
  cout << "-----\nK=" <<i-1 << "\n";
  cout << "Sum=" << K;
}
else
{
 cout << "Error";   
}
}
