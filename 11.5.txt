//Даны целые положительные числа A и B. Найти их наибольший общий делитель (НОД), используя алгоритм Евклида
#include <iostream>
#include <string>
using namespace std;
int main()
{
  int a, b;
  cout << "a=";
  cin >> a;
  cout << "b=";
  cin >> b;
while ((a != 0) && (b != 0))
{
    if (a > b)
    {
        a = a % b;
    }
    else
    {
        b = b % a;
    }
}
cout << "NOD=" << a + b;
}
